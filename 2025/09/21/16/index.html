C语言基础知识

C语言是一门通用的、过程化的计算机程序设计语言，由丹尼斯·里奇在1972年为了重写UNIX操作系统而开发。C语言具有高效、灵活、功能丰富、表达力强和较高的移植性等特点。

1. 基本语法结构

Hello World程序 

1
2
3
4
5

7

#include <stdio.h>

int main()
{
    printf("Hello,World!\n");
    return 0;
}

程序结构说明

  • #include <stdio.h>: 预处理指令，包含标准输入输出库
  • int main(): 主函数，程序执行的入口点
  • printf(): 标准输出函数
  • return 0: 返回值，表示程序正常结束

// 计算阶乘
int factorial(int n)
{
    if (n<=1)
    {
        return 1;
    }
    return n*factorial(n-1);
}

int main()
{
    printf("5! = %d\n", factorial(5));  // 输出: 120
    return 0;
}

}

#include <stdio.h>
#include <string.h>

// 定义学生结构体
struct Student {
    char name[50];
    int age;
    float score;
};

int main() {
    // 创建结构体变量
    struct Student student1;

    // 赋值
    strcpy(student1.name, "李四");
    student1.age=20;
    student1.score=95.5;

    // 输出
    printf("姓名: %s\n", student1.name);
    printf("年龄: %d\n", student1.age);
    printf("分数: %.1f\n", student1.score);

    return 0;
}

#include <stdio.h>

int main() {
    char operator;
    double num1, num2, result;

    printf("请输入运算符 (+, -, *, /): ");
    scanf("%c", &operator);

    printf("请输入两个数字: ");
    scanf("%lf %lf", &num1, &num2);

    switch (operator) {
        case '+':
            result=num1+num2;
            break;
        case '-':
            result=num1-num2;
            break;
        case '*':
            result=num1*num2;
            break;
        case '/':
            if (num2!=0) {
                result=num1/num2;
            } else {
                printf("错误: 除数不能为0\n");
                return 1;
            }
            break;
        default:
            printf("错误: 无效的运算符\n");
            return 1;
    }

    printf("%.2lf %c %.2lf = %.2lf\n", num1, operator, num2, result);

    return 0;
}
